# Cross-Quarter Re-estimate — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Дать PM возможность вручную переоценить инициативу в конкретном сценарии — per-allocation override 4 цифр (analyst/dev/qa/opo), не трогая Jira-поля и сохраняя одобренные сценарии.

**Architecture:** 4 новых nullable Float поля на `ScenarioAllocation` + те же 4 на `ScenarioAllocationSnapshot`. Хелпер `effective_estimate_hours()` — единый источник часов для всех потребителей. Континуэйшн-сервис считает «уже списано» по ворклогам до начала квартала и поднимает флаг ⚠. PATCH-эндпоинт для override + GET батч-эндпоинт для континуэйшн-инфо. UI: AntD Popover на кнопке «✎» в строке allocation.

**Tech Stack:** SQLAlchemy 2.0 + Alembic (batch SQLite), FastAPI, React 19 + AntD 6, React Query.

**Spec:** [docs/superpowers/specs/2026-05-11-cross-quarter-reestimate-design.md](../specs/2026-05-11-cross-quarter-reestimate-design.md)

```python
"""Effective per-role часы allocation: override если задан, иначе BacklogItem."""
from __future__ import annotations

from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from app.models import ScenarioAllocation


_OVERRIDE_FIELDS = (
    "override_estimate_analyst_hours",
    "override_estimate_dev_hours",
    "override_estimate_qa_hours",
    "override_estimate_opo_hours",
)


def has_override(allocation: "ScenarioAllocation") -> bool:
    """True если хотя бы одно из 4 override-полей не NULL."""
    return any(getattr(allocation, f) is not None for f in _OVERRIDE_FIELDS)


def effective_estimate_hours(allocation: "ScenarioAllocation") -> dict[str, float]:
    """Per-role часы allocation: override приоритетнее BacklogItem.

    Правило inherit:
      - has_override(allocation) is True  → берём 4 цифры из override (NULL → 0.0)
      - has_override(allocation) is False → берём из allocation.backlog_item.estimate_*
    """
    if has_override(allocation):
        return {
            "analyst": float(allocation.override_estimate_analyst_hours or 0.0),
            "dev": float(allocation.override_estimate_dev_hours or 0.0),
            "qa": float(allocation.override_estimate_qa_hours or 0.0),
            "opo": float(allocation.override_estimate_opo_hours or 0.0),
        }
    bi = allocation.backlog_item
    return {
        "analyst": float(getattr(bi, "estimate_analyst_hours", 0) or 0.0),
        "dev": float(getattr(bi, "estimate_dev_hours", 0) or 0.0),
        "qa": float(getattr(bi, "estimate_qa_hours", 0) or 0.0),
        "opo": float(getattr(bi, "estimate_opo_hours", 0) or 0.0),
    }
```

```python
"""Расчёт «уже списано» по issue до начала квартала сценария и флага продолжения."""
from __future__ import annotations

from datetime import date, datetime
from typing import Dict

from sqlalchemy.orm import Session, joinedload

from app.models import (
    BacklogItem,
    PlanningScenario,
    ScenarioAllocation,
    Worklog,
)

# Маппинг category code → role-bucket для агрегации ворклогов.
# Совпадает с разбивкой estimate_*_hours: analyst / dev / qa / opo.
CATEGORY_TO_ROLE: dict[str, str] = {
    # analysis-семейство
    "analysis": "analyst",
    "consult": "analyst",
    # dev
    "development": "dev",
    "tech_debt": "dev",
    # qa
    "testing": "qa",
    # ОПЭ
    "ope": "opo",
    "ope_analysis": "opo",
    "ope_development": "opo",
}

QUARTER_MONTHS = {"Q1": 1, "Q2": 4, "Q3": 7, "Q4": 10, "1": 1, "2": 4, "3": 7, "4": 10}


def _quarter_start(year: int, quarter: str) -> date:
    month = QUARTER_MONTHS[str(quarter).upper().replace("Q", "Q") if str(quarter).upper().startswith("Q") else str(quarter)]
    return date(year, month, 1)


def _empty_spent() -> dict[str, float]:
    return {"analyst": 0.0, "dev": 0.0, "qa": 0.0, "opo": 0.0}


class ContinuationService:
    """Per-scenario батч-расчёт spent + флага продолжения для всех allocations."""

    def __init__(self, db: Session):
        self.db = db

    def compute_for_scenario(self, scenario_id: str) -> Dict[str, dict]:
        """Возвращает map {allocation_id: {spent, spent_total, is_continuation, jira_estimate}}."""
        scenario = self.db.get(PlanningScenario, scenario_id)
        if scenario is None:
            return {}

        q_start = _quarter_start(scenario.year, scenario.quarter)
        q_start_dt = datetime.combine(q_start, datetime.min.time())

        allocations = (
            self.db.query(ScenarioAllocation)
            .options(joinedload(ScenarioAllocation.backlog_item))
            .filter(ScenarioAllocation.scenario_id == scenario_id)
            .all()
        )

        # Соберём список issue_id, по которым нужно считать ворклоги
        issue_ids = [
            a.backlog_item.issue_id
            for a in allocations
            if a.backlog_item and a.backlog_item.issue_id
        ]

        # Один запрос за всеми ворклогами
        spent_by_issue: dict[str, dict[str, float]] = {}
        if issue_ids:
            worklogs = (
                self.db.query(Worklog)
                .filter(
                    Worklog.issue_id.in_(issue_ids),
                    Worklog.started_at < q_start_dt,
                )
                .options(joinedload(Worklog.issue))
                .all()
            )
            for w in worklogs:
                bucket = spent_by_issue.setdefault(w.issue_id, _empty_spent())
                # category из worklog или из issue (через resolver-категорию)
                cat_code = getattr(w, "assigned_category", None)
                if cat_code is None and w.issue is not None:
                    cat_code = getattr(w.issue, "assigned_category", None) or getattr(
                        w.issue, "category", None
                    )
                role = CATEGORY_TO_ROLE.get(cat_code or "", None)
                if role is None:
                    continue
                bucket[role] += float(w.hours or 0)

        result: Dict[str, dict] = {}
        for a in allocations:
            bi: BacklogItem | None = a.backlog_item
            if bi is None or bi.issue_id is None:
                spent = _empty_spent()
            else:
                spent = spent_by_issue.get(bi.issue_id, _empty_spent())
            spent_total = sum(spent.values())
            jira_est = {
                "analyst": float(getattr(bi, "estimate_analyst_hours", 0) or 0.0) if bi else 0.0,
                "dev": float(getattr(bi, "estimate_dev_hours", 0) or 0.0) if bi else 0.0,
                "qa": float(getattr(bi, "estimate_qa_hours", 0) or 0.0) if bi else 0.0,
                "opo": float(getattr(bi, "estimate_opo_hours", 0) or 0.0) if bi else 0.0,
            }
            result[a.id] = {
                "spent": spent,
                "spent_total": spent_total,
                "is_continuation": spent_total > 0,
                "jira_estimate": jira_est,
            }
        return result
```

```python
from app.schemas.scenario_override import AllocationOverrideRequest


@router.patch("/scenarios/{scenario_id}/allocations/{allocation_id}/override")
def patch_allocation_override(
    scenario_id: str,
    allocation_id: str,
    body: AllocationOverrideRequest,
    db: Session = Depends(get_db),
    bus: EventBroadcaster = Depends(get_event_bus),
):
    """Сохранить (или сбросить) override 4 ролей на allocation. 409 если сценарий approved."""
    scenario = db.get(PlanningScenario, scenario_id)
    if scenario is None:
        raise HTTPException(status_code=404, detail="scenario not found")
    if scenario.status == "approved":
        raise HTTPException(status_code=409, detail="approved scenarios are read-only")

    alloc = db.get(ScenarioAllocation, allocation_id)
    if alloc is None or alloc.scenario_id != scenario_id:
        raise HTTPException(status_code=404, detail="allocation not found in scenario")

    alloc.override_estimate_analyst_hours = body.analyst
    alloc.override_estimate_dev_hours = body.dev
    alloc.override_estimate_qa_hours = body.qa
    alloc.override_estimate_opo_hours = body.opo
    db.commit()
    db.refresh(alloc)

    bus.publish("entity_changed", {"type": "scenario_allocation", "scenario_id": scenario_id})

    return {
        "id": alloc.id,
        "scenario_id": alloc.scenario_id,
        "backlog_item_id": alloc.backlog_item_id,
        "override_estimate_analyst_hours": alloc.override_estimate_analyst_hours,
        "override_estimate_dev_hours": alloc.override_estimate_dev_hours,
        "override_estimate_qa_hours": alloc.override_estimate_qa_hours,
        "override_estimate_opo_hours": alloc.override_estimate_opo_hours,
    }
```

## Task 15: AllocationOverridePopover

**Files:**
- Create: `frontend/src/components/planning/AllocationOverridePopover.tsx`

- [ ] **Step 1: Компонент**

```tsx
import { useState, useMemo } from "react";
import { Popover, Button, Table, InputNumber, Space, Typography } from "antd";
import { EditOutlined } from "@ant-design/icons";
import type { ContinuationInfoRow } from "../../api/planning";
import { useAllocationOverrideMutation } from "../../hooks/usePlanning";

const { Text } = Typography;

interface Props {
  scenarioId: string;
  allocationId: string;
  scenarioStatus: "draft" | "approved";
  currentOverride: {
    analyst: number | null;
    dev: number | null;
    qa: number | null;
    opo: number | null;
  };
  continuation: ContinuationInfoRow | undefined;
}

const ROLE_LABEL: Record<string, string> = {
  analyst: "Аналитика",
  dev: "Разработка",
  qa: "Тестирование",
  opo: "ОПЭ",
};

export function AllocationOverridePopover({
  scenarioId,
  allocationId,
  scenarioStatus,
  currentOverride,
  continuation,
}: Props) {
  const [open, setOpen] = useState(false);

  const initial = useMemo(() => {
    const jira = continuation?.jira_estimate ?? { analyst: 0, dev: 0, qa: 0, opo: 0 };
    const spent = continuation?.spent ?? { analyst: 0, dev: 0, qa: 0, opo: 0 };
    return {
      analyst: currentOverride.analyst ?? Math.max(0, (jira.analyst ?? 0) - (spent.analyst ?? 0)),
      dev: currentOverride.dev ?? Math.max(0, (jira.dev ?? 0) - (spent.dev ?? 0)),
      qa: currentOverride.qa ?? Math.max(0, (jira.qa ?? 0) - (spent.qa ?? 0)),
      opo: currentOverride.opo ?? Math.max(0, (jira.opo ?? 0) - (spent.opo ?? 0)),
    };
  }, [currentOverride, continuation]);

  const [values, setValues] = useState(initial);
  const mut = useAllocationOverrideMutation(scenarioId);

  const hasOverride =
    currentOverride.analyst !== null ||
    currentOverride.dev !== null ||
    currentOverride.qa !== null ||
    currentOverride.opo !== null;

  const handleSave = () => {
    mut.mutate(
      { allocationId, payload: values },
      { onSuccess: () => setOpen(false) },
    );
  };

  const handleReset = () => {
    mut.mutate(
      { allocationId, payload: { analyst: null, dev: null, qa: null, opo: null } },
      { onSuccess: () => setOpen(false) },
    );
  };

  const spent = continuation?.spent ?? { analyst: 0, dev: 0, qa: 0, opo: 0 };
  const jira = continuation?.jira_estimate ?? { analyst: 0, dev: 0, qa: 0, opo: 0 };

  const content = (
    <div style={{ minWidth: 380 }}>
      <Text type="secondary" style={{ display: "block", fontSize: 12, marginBottom: 4 }}>
        Оригинал Jira: А {jira.analyst} / Р {jira.dev} / Т {jira.qa} / ОПЭ {jira.opo} ={" "}
        {jira.analyst + jira.dev + jira.qa + jira.opo}ч
      </Text>
      <Text type="secondary" style={{ display: "block", fontSize: 12, marginBottom: 12 }}>
        Списано в прошлых периодах: А {spent.analyst} / Р {spent.dev} / Т {spent.qa} / ОПЭ {spent.opo} ={" "}
        {(continuation?.spent_total ?? 0)}ч
      </Text>

      <Table
        size="small"
        pagination={false}
        rowKey="role"
        dataSource={(["analyst", "dev", "qa", "opo"] as const).map((role) => ({
          role,
          label: ROLE_LABEL[role],
          value: values[role],
          spent: spent[role],
          remainder: Math.max(0, (jira[role] ?? 0) - (spent[role] ?? 0)),
        }))}
        columns={[
          { title: "Роль", dataIndex: "label" },
          {
            title: "План Q",
            dataIndex: "value",
            render: (_, row) => (
              <div>
                <InputNumber
                  min={0}
                  value={row.value}
                  onChange={(v) => setValues((s) => ({ ...s, [row.role]: v ?? 0 }))}
                  style={{ width: 80 }}
                />
                {row.value !== null && row.value < row.spent && (
                  <Text type="danger" style={{ display: "block", fontSize: 11 }}>
                    план меньше уже списанного
                  </Text>
                )}
              </div>
            ),
          },
          { title: "Списано", dataIndex: "spent", align: "right" as const },
          { title: "Остаток оригинала", dataIndex: "remainder", align: "right" as const },
        ]}
      />

      <Space style={{ marginTop: 12, justifyContent: "flex-end", display: "flex", width: "100%" }}>
        {hasOverride && (
          <Button onClick={handleReset} disabled={scenarioStatus === "approved"}>
            Сбросить
          </Button>
        )}
        <Button
          type="primary"
          onClick={handleSave}
          loading={mut.isPending}
          disabled={scenarioStatus === "approved"}
        >
          Сохранить
        </Button>
      </Space>
    </div>
  );

  return (
    <Popover
      open={open}
      onOpenChange={setOpen}
      trigger="click"
      content={content}
      placement="bottomRight"
    >
      <Button size="small" icon={<EditOutlined />} />
    </Popover>
  );
}
```
